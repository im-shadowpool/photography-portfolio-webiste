    <script>
        const track = document.getElementById('stripTrack');
        const container = document.getElementById('stripContainer');
        const handle = document.getElementById('handle');
        
        let isDown = false;
        let startX;
        let scrollLeft;

        container.addEventListener('mousedown', (e) => {
            isDown = true;
            startX = e.pageX - container.offsetLeft;
            scrollLeft = container.scrollLeft;
            container.style.cursor = 'grabbing';
        });

        container.addEventListener('mouseleave', () => {
            isDown = false;
            container.style.cursor = 'grab';
        });

        container.addEventListener('mouseup', () => {
            isDown = false;
            container.style.cursor = 'grab';
        });

        container.addEventListener('mousemove', (e) => {
            if (!isDown) return;
            e.preventDefault();
            const x = e.pageX - container.offsetLeft;
            const walk = (x - startX) * 2;
            container.scrollLeft = scrollLeft - walk;
            updateScrubber();
        });

        // Wheel horizontal scroll
        container.addEventListener('wheel', (e) => {
            if (e.deltaY !== 0) {
                e.preventDefault();
                container.scrollLeft += e.deltaY;
                updateScrubber();
            }
        });

        function updateScrubber() {
            const maxScroll = container.scrollWidth - container.clientWidth;
            const scrollPercentage = (container.scrollLeft / maxScroll) * 100;
            handle.style.left = `${Math.min(Math.max(scrollPercentage, 0), 90)}%`;
        }

        // Entrance Animation
        const observerOptions = {
            threshold: 0.1
        };

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const frames = entry.target.querySelectorAll('.film-frame');
                    frames.forEach((frame, index) => {
                        setTimeout(() => {
                            frame.style.opacity = '1';
                            frame.style.transform = 'translateX(0)';
                        }, index * 150);
                    });
                }
            });
        }, observerOptions);

        observer.observe(container);

        // Set initial state for animation
        document.querySelectorAll('.film-frame').forEach(f => {
            f.style.opacity = '0';
            f.style.transform = 'translateX(50px)';
            f.style.transition = 'all 0.8s cubic-bezier(0.23, 1, 0.32, 1)';
        });
    </script>
