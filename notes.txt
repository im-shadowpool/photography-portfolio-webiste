    <script>
        gsap.registerPlugin(ScrollTrigger);

        const container = document.querySelector(".horizontal-container");
        const cards = gsap.utils.toArray(".service-card");

        // Horizontal Scroll Animation
        let scrollTween = gsap.to(container, {
            x: () => -(container.scrollWidth - document.documentElement.clientWidth) + "px",
            ease: "none",
            scrollTrigger: {
                trigger: ".scroll-section",
                pin: true,
                scrub: 1,
                end: () => "+=" + container.scrollWidth,
                onUpdate: (self) => {
                    gsap.to(".progress-bar", { width: `${self.progress * 100}%`, duration: 0.1 });
                }
            }
        });

        // Individual Card Reveals
        cards.forEach((card, i) => {
            gsap.fromTo(card, 
                { opacity: 0, y: 50, scale: 0.95 }, 
                { 
                    opacity: 1, 
                    y: 0, 
                    scale: 1,
                    duration: 0.8,
                    ease: "power2.out",
                    scrollTrigger: {
                        trigger: card,
                        containerAnimation: scrollTween,
                        start: "left 90%",
                        toggleActions: "play none none reverse"
                    }
                }
            );

            // Light-streak effect on scroll
            gsap.fromTo(card.querySelector('.card-visual img'), 
                { x: -20 },
                { 
                    x: 20,
                    ease: "none",
                    scrollTrigger: {
                        trigger: card,
                        containerAnimation: scrollTween,
                        scrub: true
                    }
                }
            );
        });

        // Section Title Parallax
        gsap.to(".section-title h2", {
            x: -200,
            opacity: 1,
            scrollTrigger: {
                trigger: ".scroll-section",
                scrub: true,
                start: "top top"
            }
        });
    </script>
